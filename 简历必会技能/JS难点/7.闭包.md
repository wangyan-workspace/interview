## 闭包的三个特性
1、函数嵌套函数；
2、函数内部可以访问函数外部的参数和变量；
3、闭包内部的变量不会被垃圾回收；
## 使用场景
任何闭包的使用场景都离不开这两点：   
1.创建私有变量   
2.延长变量的生命周期     
一般函数的词法环境在函数返回后就被销毁，但是闭包会保存对创建时所在词法环境的引用，即便创建时所在的执行上下文被销毁，但创建时所在词法环境依然存在，以达到延长变量的生命周期的目的   
## 例子
``` 
    如果函数作用域中的变量或函数未被引用，则用完函数作用域立即销毁    
    所以将sum函数返回出去（给外界使用），则函数不会被销毁，通过sum函数操作n，所以n会被保留    
    function hd1() {
        let n = 1;
        return function sum() {
            console.log(++n);
        }
    }
    // 函数被调用会开辟新的空间，互不影响
    let a = hd1();
    a();//2
    a();//3
    a();//4
    a();//5
    let b = hd1();
    b();//2
    b();//3
    b();//4
    b();//5
```

```
    function hd2() {
        let n = 1;
        return function sum() {
            // console.log('n:' + ++n);
            let m = 1;
            function show() {
                let m = 1;
                console.log('n:' + ++n);//sum函数返回给外界，外界会用到，n是父函数中的变量，会被保留
                console.log('m:' + ++m); //执行完自动销毁不会被保留
            }
            show();
        }
    }

    let c = hd2();
    c(); //n:2 m:2
    c(); //n:3 m:2
    c(); //n:4 m:2
```

```
    // 延长函数的生命周期
    function hd3() {
        let n = 1;
        return function sum() {
            // console.log('n:' + ++n);
            let m = 1;
            return function show() {
                console.log('n:' + ++n);
                console.log('m:' + ++m);//被外界用到所以会被保存
            }
        }
    }

    let d = hd3()();
    d(); //n:2 m:2
    d(); //n:3 m:3
    d(); //n:4 m:4
```

```
    //构造函数中的作用域的使用形态
    function Hd() {
        let n=1;
        this.sum = function () {
            console.log(++n);  //n被用到了所以会被保留
        }
    }
    // 相当于
    function Hd() {
        let n = 1;
        function sum() {
            console.log(++n);
        }
        return { sum: sum }
    }
    let e = new Hd();
    console.log(e);
    e.sum(); //2
    e.sum(); //3
    let f = new Hd();
    console.log(f);
    f.sum(); //2
    f.sum(); //3
```

```
    // 使用闭包获取区间商品
    let arr = [1, 23, 4, 5, 6, 7, 8, 9, 21, 10];
    function between(a, b) {
        return function (item) {
            return item >= a && item <= b;
        };
    }
    //这里涉及到回调函数的概念，先看filter函数，
    //这个函数接收一个回调函数，这个回调函数接受v作为参数。
    //在filter遍历的时候会调用这个回调函数并且把每项值v传给这个回调函数。
    let arrs = arr.filter(between(3, 9));
    console.log(arrs);  //(6) [4, 5, 6, 7, 8, 9]
```
## 闭包用法
代码模块化 避免全局污染
循环中找到对应的元素的索引，不需要加索引（let最方便）
```
    var ALi = document.getElementsByTagName("li");
    console.log(ALi);  //HTMLCollection(8) [li, li, li, li, li, li, li, li]
    for (var i = 0; i < ALi.length; i++) {
        ALi[i].onclick = function(){
            console.log(i); //8
        }
    }
    // let解决
    var ALi = document.getElementsByTagName("li");
    console.log(ALi);  //HTMLCollection(8) [li, li, li, li, li, li, li, li]
    for (let i = 0; i < ALi.length; i++) {
        ALi[i].onclick = function(){
            console.log(i); //0 1 2 3 4 5 6 7
        }
    }
    // 闭包解决
    var ALi = document.getElementsByTagName("li");
    console.log(ALi);  //HTMLCollection(8) [li, li, li, li, li, li, li, li]
    for (let i = 0; i < ALi.length; i++) {
        (function (i) {
            ALi[i].onclick = function () {
                console.log(i); //0 1 2 3 4 5 6 7
            }
        })(i)
    }
```
## 闭包的好处
1.变量长期驻扎在内存中
2.避免全局变量污染（代码模块化）
3.私有成员的存在
## 需要注意的是
1.会引发内存泄露  
如果不是某些特定任务需要使用闭包，在其它函数中创建函数是不明智的，因为闭包在处理速度和内存消耗方面对脚本性能具有负面影响

例如，在创建新的对象或者类时，方法通常应该关联于对象的原型，而不是定义到对象的构造器中。

原因在于每个对象的创建，方法都会被重新赋值

```
    function MyObject(name, message) {
    this.name = name.toString();
    this.message = message.toString();
    this.getName = function() {
        return this.name;
    };

    this.getMessage = function() {
        return this.message;
    };
    }
```
上面的代码中，我们并没有利用到闭包的好处，因此可以避免使用闭包。修改成如下：  
```
    function MyObject(name, message) {
    this.name = name.toString();
    this.message = message.toString();
    }
    MyObject.prototype.getName = function() {
    return this.name;
    };
    MyObject.prototype.getMessage = function() {
    return this.message;
    };
```
